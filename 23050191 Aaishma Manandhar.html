# Step 1: Convert date columns to datetime format
df['Created Date'] = pd.to_datetime(df['Created Date'], format='mixed')
df['Closed Date'] = pd.to_datetime(df['Closed Date'], format='mixed')

# Step 2: Calculate request closing time in hours
df['Request_Closing_Hours'] = (df['Closed Date'] - df['Created Date']).dt.total_seconds() / 3600

# Step 3: Drop rows with missing values
df = df.dropna(subset=['Complaint Type', 'Borough', 'Request_Closing_Hours'])

# Step 4: Group by borough and complaint type to get average closing time
grouped = df.groupby(['Borough', 'Complaint Type'])['Request_Closing_Hours'].mean().reset_index()

# Step 5: Pivot for plotting
pivot_df = grouped.pivot(index='Complaint Type', columns='Borough', values='Request_Closing_Hours')

# Step 6: Plot
pivot_df.plot(kind='bar', figsize=(12, 6))
plt.title('Average Request Closing Time by Complaint Type Across Boroughs')
plt.xlabel('Complaint Type')
plt.ylabel('Avg. Closing Time (Hours)')
plt.xticks(rotation=45)
plt.legend(title='Borough')
plt.tight_layout()
plt.show()
